To install a clean version of the repo, you must satisfy the following requirements/commands:

0- sudo apt-get install python3-pip
1- pip install deepface
2- pip install django
3- pip install django-sage-streaming
4- pip install django-multiselectfield

5- sudo apt-get install ros-noetic-web-video-server
6- sudo apt-get install ros-noetic-usb-cam
7- sudo apt-get install ros-noetic-audio-common

https://github.com/CircusMonkey/ros_rtsp:
    8- sudo apt-get install libgstreamer-plugins-base1.0-dev libgstreamer-plugins-good1.0-dev libgstreamer-plugins-bad1.0-dev libgstrtspserver-1.0-dev gstreamer1.0-plugins-ugly gstreamer1.0-plugins-bad
    8-1- git clone https://github.com/CircusMonkey/ros_rtsp.git
    8-2- cd ..
    8-3- catkin_make pkg:=ros_rtsp
    8-4- Change the config/stream_setup.yaml to suit your required streams.

9- pip install transformers
