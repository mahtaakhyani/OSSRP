37def get_headers_and_data(data: Union[str, bytes]) -> Tuple[Dict[bytes, bytes], bytes]:
38 """
39 Returns the headers and data from the given data.
40
41 Args:
42 data (str or bytes): The data to be parsed.
43
44 Returns:
45 tuple: The headers and data to be used in the request.
46 """
47 if isinstance(data, str):
48 data = data.encode("utf-8")
49 if not isinstance(data, bytes):
50 raise TypeError("data must be str or bytes")
51
52 headers = {}
53 for line in data[: data.find(b"\r\n\r\n")].split(b"\r\n"):
54 key, value = line.split(b":", 1)
55 headers[key] = value
56
57 return headers, data[data.find(b"\r\n\r\n") + 4 :]

60def remove_incompatible_characters(string: Union[str, bytes]) -> str:
61 """
62 The service does not support a couple character ranges.
63 Most important being the vertical tab character which is
64 commonly present in OCR-ed PDFs. Not doing this will
65 result in an error from the service.
66
67 Args:
68 string (str or bytes): The string to be cleaned.
69
70 Returns:
71 str: The cleaned string.
72 """
73 if isinstance(string, bytes):
74 string = string.decode("utf-8")
75 if not isinstance(string, str):
76 raise TypeError("string must be str or bytes")
77
78 chars: List[str] = list(string)
79
80 for idx, char in enumerate(chars):
81 code: int = ord(char)
82 if (0 <= code <= 8) or (11 <= code <= 12) or (14 <= code <= 31):
83 chars[idx] = " "
84
85 return "".join(chars)

98def split_text_by_byte_length(
99 text: Union[str, bytes], byte_length: int
100) -> Generator[bytes, None, None]:
101 """
102 Splits a string into a list of strings of a given byte length
103 while attempting to keep words together. This function assumes
104 text will be inside of an XML tag.
105
106 Args:
107 text (str or bytes): The string to be split.
108 byte_length (int): The maximum byte length of each string in the list.
109
110 Yield:
111 bytes: The next string in the list.
112 """
113 if isinstance(text, str):
114 text = text.encode("utf-8")
115 if not isinstance(text, bytes):
116 raise TypeError("text must be str or bytes")
117
118 if byte_length <= 0:
119 raise ValueError("byte_length must be greater than 0")
120
121 while len(text) > byte_length:
122 # Find the last space in the string
123 split_at = text.rfind(b" ", 0, byte_length)
124
125 # If no space found, split_at is byte_length
126 split_at = split_at if split_at != -1 else byte_length
127
128 # Verify all & are terminated with a ;
129 while b"&" in text[:split_at]:
130 ampersand_index = text.rindex(b"&", 0, split_at)
131 if text.find(b";", ampersand_index, split_at) != -1:
132 break
133
134 split_at = ampersand_index - 1
135 if split_at < 0:
136 raise ValueError("Maximum byte length is too small or invalid text")
137 if split_at == 0:
138 break
139
140 # Append the string to the list
141 new_text = text[:split_at].strip()
142 if new_text:
143 yield new_text
144 if split_at == 0:
145 split_at = 1
146 text = text[split_at:]
147
148 new_text = text.strip()
149 if new_text:
150 yield new_text

171def date_to_string() -> str:
172 """
173 Return Javascript-style date string.
174
175 Returns:
176 str: Javascript-style date string.
177 """
178 # %Z is not what we want, but it's the only way to get the timezone
179 # without having to use a library. We'll just use UTC and hope for the best.
180 # For example, right now %Z would return EEST when we need it to return
181 # Eastern European Summer Time.
182 return time.strftime(
183 "%a %b %d %Y %H:%M:%S GMT+0000 (Coordinated Universal Time)", time.gmtime()

204def calc_max_mesg_size(voice: str, rate: str, volume: str) -> int:
205 """Calculates the maximum message size for the given voice, rate, and volume.
206
207 Returns:
208 int: The maximum message size.
209 """
210 websocket_max_size: int = 2**16
211 overhead_per_message: int = (
212 len(
213 ssml_headers_plus_data(
214 connect_id(),
215 date_to_string(),
216 mkssml("", voice, rate, volume),
217 )
218 )
219 + 50 # margin of error
220 )
221 return websocket_max_size - overhead_per_message

293 async def stream(self) -> AsyncGenerator[Dict[str, Any], None]:
294 """Streams audio and metadata from the service."""
295
296 texts = split_text_by_byte_length(
297 escape(remove_incompatible_characters(self.text)),
298 calc_max_mesg_size(self.voice, self.rate, self.volume),
299 )
300 final_utterance: Dict[int, int] = {}
301 prev_idx = -1
302 shift_time = -1
303
304 for idx, text in enumerate(texts):
305 async with aiohttp.ClientSession(
306 trust_env=True
307 ) as session, session.ws_connect(
308 f"{WSS_URL}&ConnectionId={connect_id()}",
309 compress=15,
310 autoclose=True,
311 autoping=True,
312 proxy=self.proxy,
313 headers={
314 "Pragma": "no-cache",
315 "Cache-Control": "no-cache",
316 "Origin": "chrome-extension://jdiccldimpdaibmpdkjnbmckianbfold",
317 "Accept-Encoding": "gzip, deflate, br",
318 "Accept-Language": "en-US,en;q=0.9",
319 "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36"
320 " (KHTML, like Gecko) Chrome/91.0.4472.77 Safari/537.36 Edg/91.0.864.41",
321 },
322 ) as websocket:
323 # download indicates whether we should be expecting audio data,
324 # this is so what we avoid getting binary data from the websocket
325 # and falsely thinking it's audio data.
326 download_audio = False
327
328 # audio_was_received indicates whether we have received audio data
329 # from the websocket. This is so we can raise an exception if we
330 # don't receive any audio data.
331 audio_was_received = False
332
333 # Each message needs to have the proper date.
334 date = date_to_string()
335
336 # Prepare the request to be sent to the service.
337 #
338 # Note sentenceBoundaryEnabled and wordBoundaryEnabled are actually supposed
339 # to be booleans, but Edge Browser seems to send them as strings.
340 #
341 # This is a bug in Edge as Azure Cognitive Services actually sends them as
342 # bool and not string. For now I will send them as bool unless it causes
343 # any problems.
344 #
345 # Also pay close attention to double { } in request (escape for f-string).
346 await websocket.send_str(
347 f"X-Timestamp:{date}\r\n"
348 "Content-Type:application/json; charset=utf-8\r\n"
349 "Path:speech.config\r\n\r\n"
350 '{"context":{"synthesis":{"audio":{"metadataoptions":{'
351 '"sentenceBoundaryEnabled":false,"wordBoundaryEnabled":true},'
352 '"outputFormat":"audio-24khz-48kbitrate-mono-mp3"'
353 "}}}}\r\n"
354 )
355
356 await websocket.send_str(
357 ssml_headers_plus_data(
358 connect_id(),
359 date,
360 mkssml(text, self.voice, self.rate, self.volume),
361 )
362 )
363
364 async for received in websocket:
365 if received.type == aiohttp.WSMsgType.TEXT:
366 parameters, data = get_headers_and_data(received.data)
367 path = parameters.get(b"Path")
368 if path == b"turn.start":
369 download_audio = True
370 elif path == b"turn.end":
371 download_audio = False
372 break # End of audio data
373 elif path == b"audio.metadata":
374 for meta_obj in json.loads(data)["Metadata"]:
375 meta_type = meta_obj["Type"]
376 if idx != prev_idx:
377 shift_time = sum(
378 final_utterance[i] for i in range(idx)
379 )
380 prev_idx = idx
381 if meta_type == "WordBoundary":
382 final_utterance[idx] = (
383 meta_obj["Data"]["Offset"]
384 + meta_obj["Data"]["Duration"]
385 # Average padding added by the service
386 # Alternatively we could use ffmpeg to get value properly
387 # but I don't want to add an additional dependency
388 # if this is found to work well enough.
389 + 8_750_000
390 )
391 yield {
392 "type": meta_type,
393 "offset": meta_obj["Data"]["Offset"]
394 + shift_time,
395 "duration": meta_obj["Data"]["Duration"],
396 "text": meta_obj["Data"]["text"]["Text"],
397 }
398 elif meta_type == "SessionEnd":
399 continue
400 else:
401 raise UnknownResponse(
402 f"Unknown metadata type: {meta_type}"
403 )
404 elif path == b"response":
405 pass
406 else:
407 raise UnknownResponse(
408 "The response from the service is not recognized.\n"
409 + received.data
410 )
411 elif received.type == aiohttp.WSMsgType.BINARY:
412 if not download_audio:
413 raise UnexpectedResponse(
414 "We received a binary message, but we are not expecting one."
415 )
416
417 yield {
418 "type": "audio",
419 "data": received.data[
420 received.data.find(b"Path:audio\r\n") + 12 :
421 ],
422 }
423 audio_was_received = True
424 elif received.type == aiohttp.WSMsgType.ERROR:
425 raise WebSocketError(
426 received.data if received.data else "Unknown error"
427 )
428
429 if not audio_was_received:
430 raise NoAudioReceived(
431 "No audio was received. Please verify that your parameters are correct."
