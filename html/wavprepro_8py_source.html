47def get_spectrograms(fpath):
48

51
52 # Trimming
53 y, _ = librosa.effects.trim(y)

56 y = np.append(y[0], y[1:] - hp.preemphasis * y[:-1])

59 linear = librosa.stft(y=y,

61 hop_length=hp.hop_length,

65 mag = np.abs(linear) # (1+n_fft//2, T)

67 # mel spectrogram

69 mel = np.dot(mel_basis, mag) # (F, t)

110 X_best = copy.deepcopy(spectrogram)
