93 @staticmethod
94 def get_square_box(box):
95 """Get a square box out of the given box, by expanding it."""
96 left_x = box[0]
97 top_y = box[1]
98 right_x = box[2]
99 bottom_y = box[3]
100
101 box_width = right_x - left_x
102 box_height = bottom_y - top_y
103
104 # Check if box is already a square. If not, make it a square.
105 diff = box_height - box_width
106 delta = int(abs(diff) / 2)
107
108 if diff == 0: # Already a square.
109 return box
110 elif diff > 0: # Height > width, a slim box.
111 left_x -= delta
112 right_x += delta
113 if diff % 2 == 1:
114 right_x += 1
115 else: # Width > height, a short box.
116 top_y -= delta
117 bottom_y += delta
118 if diff % 2 == 1:
119 bottom_y += 1
120
121 # Make sure box is always square.
122 assert ((right_x - left_x) == (bottom_y - top_y)), 'Box is not square.'
123
124 return [left_x, top_y, right_x, bottom_y]
